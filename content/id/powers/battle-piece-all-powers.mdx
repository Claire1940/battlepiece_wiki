  </Step>
  <Step num="3" title="Dorong Menuju Batas Level 600">

  </Step>
  <Step num="4" title="Buka dan Taklukkan Raid yang Menantang">

  </Step>
  <Step num="5" title="Uji Setup Anda dalam Pertarungan PvP">
